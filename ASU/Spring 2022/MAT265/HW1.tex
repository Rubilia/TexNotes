\begin{task}[2]
$f(x) = \left\{
\begin{array}{ll}
x^2-x-3 &\text{if }x < -2\\
-1 & \text{if }x = -2\\
x^2-1 & \text{if }(-2<x)\text{ and }(x < 1)\\
0 & \text{if } x=1\\
3x^2-2x-1 & \text{if }(1<x)\text{ and }(x < 2)\\
7 & \text{if } x=2\\
x^2+3x & \text{if } 2<x\\
\end{array}
\right.$
\newpage
\begin{description}
	\item[1. $a=-2$]{
$\\\lim\limits_{x \to -2^{-}} \underbrace{f(x) = \lim\limits_{x \to -2^{-}} (x^2-x-3)}_{f(x)=x^2-x-3\text{ for }x <-2}=(4-(-2)-3)=3\\
\lim\limits_{x \to -2^{+}}\underbrace{f(x)=\lim\limits_{x \to -2^{+}} (x^2-1)}_{f(x)=x^2-1\text{ for }x>-2}=4-1=3\\
\left.
\begin{array}{l}
	\lim\limits_{x \to -2^{-}} f(x) =3\\
	\lim\limits_{x \to -2^{+}} f(x)=3\\
\end{array}
\right\} \implies \lim\limits_{x \to -2}f(x)=3\\
f(-2)=-1 \neq \lim\limits_{x \to -2}f(x) \implies f(x)$ is not continuous at $x=-2$
}
\item[2. $a=1$]{
$\\\lim\limits_{x \to 1^{-}} \underbrace{f(x) = \lim\limits_{x \to 1^{-}} (x^2-1)}_{f(x)=x^2-1\text{ for } x < 1}=1-1=0\\
\lim\limits_{x \to 1^{+}} \underbrace{f(x) = \lim\limits_{x \to 1^{+}} (3x^2-2x-1)}_{f(x) = 3x^2-2x-1\text{ for } x > 1}=3-2-1=0\\
\left.
\begin{array}{l}
\lim\limits_{x \to 1^{-}} f(x)=0\\
\lim\limits_{x \to 1^{+}} f(x)=0\\
\end{array}
\right\} \implies \lim\limits_{x \to 1} f(x)=0\\
f(1)=\lim\limits_{x \to 1} f(x) \implies f(x)$ is continuous at $x=1$
}
\item[3. $a=2$]{
$\\\lim\limits_{x \to 2^{-}}\underbrace{f(x) = \lim\limits_{x \to 2^{-}} (3x^2-2x-1)}_{f(x)=3x^2-2x-1\text{ for } x < 2}=3 \cdot 2^2 - 2 \cdot 2 - 1 = 7\\
\lim\limits_{x \to 2^{+}} \underbrace{f(x) = \lim\limits_{x \to 2^{+}} (x^2+3x)}_{f(x)=x^2+3x\text{ for } x>2}=2^2+3 \cdot 2=10\\
\left.
\begin{array}{l}
\lim\limits_{x \to 2^{-}} f(x) = 7\\
\lim\limits_{x \to 2^{+}} f(x) = 10\\
\end{array}
\right\} \implies \lim\limits_{x \to 2} f(x) $   DNE\\
Since $\lim\limits_{x \to 2} f(x)$ DNE $\implies  \enspace f(x)$ is not continuous at x = 2\\ 
}
\end{description}
\end{task}
